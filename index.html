async function loadAux(){

  const map = {

    taxonomia:"aux_taxonomia", ods:"aux_ods", lai_motivos:"aux_lai_motivos",

    servidores:"aux_servidores", unidades:"aux_unidades", orgaos:"aux_orgaos",

    tipos_demanda:"aux_tipos_demanda", tipos_documento:"aux_tipos_documento", status:"aux_status"

  };

  for(const [k,tbl] of Object.entries(map)){

    try{

      const {data,error}=await sb.from(tbl).select("*");

      if(error)throw error;

      if(data && data.length){

        if(k==="tipos_demanda"||k==="tipos_documento"||k==="status") AUX[k]=data.map(r=>r.nome);

        else AUX[k]=data;

      }else AUX[k]=fallback(k);

    }catch(e){AUX[k]=fallback(k);}

  }

}

function fallback(k){

  if(k==="taxonomia")return SEED.taxonomia;

  if(k==="ods")return SEED.ods;

  if(k==="lai_motivos")return SEED.lai_motivos;

  if(k==="servidores")return SEED.servidores;

  if(k==="unidades")return SEED.unidades;

  if(k==="orgaos")return SEED.orgaos;

  if(k==="tipos_demanda")return SEED.tipos_demanda;

  if(k==="tipos_documento")return SEED.tipos_documento;

  if(k==="status")return SEED.status;

  return [];

}
